API Endpoints

POST /generateSpec

Converts an Ethereum ABI JSON into an OpenAPI specification.

Request

Content-Type: application/json

Body: Ethereum ABI JSON array

[
  {
    "inputs": [],
    "name": "totalSupply",
    "outputs": [
      {
        "internalType": "uint256",
        "name": "",
        "type": "uint256"
      }
    ],
    "stateMutability": "view",
    "type": "function"
  },
  {
    "inputs": [
      {
        "internalType": "address",
        "name": "to",
        "type": "address"
      },
      {
        "internalType": "uint256",
        "name": "amount",
        "type": "uint256"
      }
    ],
    "name": "transfer",
    "outputs": [
      {
        "internalType": "bool",
        "name": "",
        "type": "bool"
      }
    ],
    "stateMutability": "nonpayable",
    "type": "function"
  }
]

Response

Content-Type: application/json

Body: OpenAPI specification in JSON format

{
  "openapi": "3.0.0",
  "info": {
    "title": "Smart Contract API",
    "version": "1.0.0",
    "description": "API for interacting with a smart contract"
  },
  "paths": {
    "/totalSupply": {
      "get": {
        "summary": "totalSupply",
        "description": "Function: totalSupply()",
        "operationId": "totalSupply",
        "responses": {
          "200": {
            "description": "Successful response",
            "content": {
              "application/json": {
                "schema": {
                  "type": "object",
                  "properties": {
                    "result": {
                      "type": "string",
                      "description": "uint256"
                    }
                  }
                }
              }
            }
          }
        }
      }
    },
    "/transfer": {
      "post": {
        "summary": "transfer",
        "description": "Function: transfer(address,uint256)",
        "operationId": "transfer",
        "requestBody": {
          "required": true,
          "content": {
            "application/json": {
              "schema": {
                "type": "object",
                "required": ["to", "amount"],
                "properties": {
                  "to": {
                    "type": "string",
                    "description": "address"
                  },
                  "amount": {
                    "type": "string",
                    "description": "uint256"
                  }
                }
              }
            }
          }
        },
        "responses": {
          "200": {
            "description": "Successful response",
            "content": {
              "application/json": {
                "schema": {
                  "type": "object",
                  "properties": {
                    "result": {
                      "type": "boolean"
                    }
                  }
                }
              }
            }
          }
        }
      }
    }
  },
  "components": {
    "schemas": {}
  }
}

Usage Examples

Curl Example

curl -X POST https://your-app-url/generateSpec \
  -H "Content-Type: application/json" \
  -d @your-abi.json

JavaScript Example

async function convertAbiToOpenApi(abiJson) {
  const response = await fetch('https://your-app-url/generateSpec', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(abiJson)
  });

  if (!response.ok) {
    throw new Error(`HTTP error! Status: ${response.status}`);
  }

  return await response.json();
}

Node.js Example

const fetch = require('node-fetch');
const fs = require('fs');

// Read ABI from file
const abiJson = JSON.parse(fs.readFileSync('path/to/your-abi.json', 'utf8'));

// Convert to OpenAPI
fetch('https://your-app-url/generateSpec', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json'
  },
  body: JSON.stringify(abiJson)
})
.then(response => {
  if (!response.ok) {
    throw new Error(`HTTP error! Status: ${response.status}`);
  }
  return response.json();
})
.then(openApiSpec => {
  console.log(JSON.stringify(openApiSpec, null, 2));
  // Optionally save to file
  fs.writeFileSync('openapi-spec.json', JSON.stringify(openApiSpec, null, 2));
})
.catch(error => console.error('Error:', error));
